using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace lab1
{
    class Program
    {
        static void Main(string[] args)
        {
            byte number1, number2;
            byte result;
            number1 = 127;
            number2 = 127;
            try
            {
                result = unchecked((byte)(number1 * number2));
                Console.WriteLine("Unchecked result:" + result);
                result = checked((byte)(number1 * number2));
                Console.WriteLine("checked result:" + result);
            }
            catch (OverflowException exc)
            {
                Console.WriteLine(exc);
            }
            finally
            {
                Console.WriteLine("press Enter to Quit..");
                Console.ReadLine();
            }
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace lab3
{
    class Mylist
    {
        string[] names;
        public Mylist()
        {
            names = new string[10];
            for (int i = 0; i < 10; i++)
            {
                names[i] = "No name";
            }
        }
        public string this[int pos]
        {
            get
            {
                return names[pos];
            }
            set
            {
                names[pos] = value;
            }
        }
        public int this[string name]
        {
            get
            {
                for (int i = 0; i < 10; i++)
                {
                    if (names[i].Equals(name))
                    {
                        return i + 1;
                    }
                }
                return -1;
            }
            set
            {
                names[value] = name;
            }
        }
    }

class Program
    {
        static void Main(string[] args)
        {
            Mylist list = new Mylist();
            list[0] = "Myname";
            list["yourname"] = 4;
            Console.WriteLine("Myname is at position:" + list["Myname"]);
            Console.WriteLine("Name at position 4 is:" + list[4]);
            Console.ReadKey();
        }
    }
}
output:
      Myname is at position:1
Name at position 4 is:yourname
